#!/bin/bash
PS3='Следующая задача: '
options=("Создать папку" "Заходим в папку" "Создаем 3 папки" "Заходим в папку" "Создаем 5 файлов" "Создаем 3 папки" "Выводим список содержимого" "Открываем текстовый файл" "Занести данные" "Выйти из папки на уровень выше" "переместить любые 2 файла, которые вы создали, в любую другую папку" "скопировать любые 2 файла, которые вы создали, в любую другую папку" "Найти файл по имени" "просмотреть содержимое в реальном времени" "вывести несколько первых строк из текстового файла" "вывести несколько последних строк из текстового файла" "просмотреть содержимое длинного файла" "вывести дату и время" "Отправить http запрос на сервер" "Выход")
select opt in "${options[@]}"
do
    case $opt in
        "Создать папку")
            echo "Для того, чтобы создать папку используется команда mkdir Имя Папки"
		echo "Введите имя папки:"
		read dirname
		mkdir $dirname
		echo "Папка с именем" $dirname "создана"
		ls -la
            ;;
        "Заходим в папку")
            echo "Для перехода в нужную папку используется команда cd Имя папки"
		echo "Введите имя папки:"
		read dirname
		cd $dirname
		echo "Мы перешли в папку" $dirname 
		pwd
            ;;
        "Создаем 3 папки")
            echo "Для создания сразу нескольких папок используется команда со следующим синтаксисом mkdir Имя папки Имя папки2 Имя папки3"
		echo "Для примера создадим 3 папки"
		echo "Введите имя папки:"
		read dirname
		echo "Введите имя папки2:"
		read dirname1
		echo "Введите имя папки3:"
		read dirname2
		mkdir $dirname $dirname1 $dirname2
		echo "Папки с именем" $dirname $dirname1 $dirname2 "созданы"
		ls -la
            ;;
	"Создаем 5 файлов")
		echo "Для создания файлов используется команда touch Имя файла.Расширение файла"
		echo "Для примера создадим 5 файлов 3 txt, 2json"
		echo "Введите имя текстового файла 1:"
		read txtname1
		echo "Введите имя текстового файла 2:"
		read txtname2
		echo "Введите имя текстового файла 3:"
		read txtname3
		echo "Введите имя js файла 1:"
		read jsname1
		echo "Введите имя js файла 2:"
		read jsname2
		touch $txtname1.txt $txtname2.txt $txtname3.txt $jsname1.js $jsname2.js
		echo "Проверяем создание файлов"
		ls -la
	;;
	"Выводим список содержимого")
		echo "Для вывода списка содержимого используется команда ls, команда ls часто используется с ключами -la ,где l указывает дополнительную информацию о правах, владельце, дате редактирования и размере, а а позволяет видеть даже скрытые файлы"
		ls -la
	;;

	"Открываем текстовый файл")
		echo "Для открытия текстового файла используется команда vim Имя файла.Расширение"
		echo "Какой файл вы хотите открыть?"
		ls -la
		read txtname
		vim $txtname.txt
	;;
	"Занести данные")
		echo "Для того, чтобы вставить текст в файл используется следующая команда echo нужный текст >>нужный файл.txt"
		
		
		echo "Для примера, введите текст который хотите сохранить"
		read newtext
		ls -la
		echo "Введите файл в который хотите сохранить введенный текст"
		read filename
		echo $newtext >>$filename.txt
		echo "Проверим что все успешно"
		vim $filename.txt
		;;
	"Выйти из папки на уровень выше")
		echo "Для того, чтобы подняться на уровень выше по каталогу используется следующая команда: cd .."
		echo "Для примера сейчас мы находимся здесь"
		pwd
		echo "используем cd .."
		cd ..
		echo "и чудесным образом мы выше"
		pwd
		;;
	"переместить любые 2 файла, которые вы создали, в любую другую папку")
		echo "Для перемещения 2 файлов используется следующая конструкция mv Имя файла.Расширение Имя файла2.Расширение  Путь куда переместить"
		echo "Давайте посмотрим есть ли что для переноса"
		ls -la
		echo "Введите 1ый файл в который хотите переместить"
		read filename
		echo "Введите 2ый файл в который хотите переместить"
		read filename1
		echo "Введите имя папки куда хотите отправить файлы:"
		read dirname	
		mv $filename $filename1 $dirname
		echo "Проверим перенесли ли мы файлы? Перейдем в нужную папку"
		cd $dirname
		ls -la
		echo "Файлы тут? отлично"
		;;
	"скопировать любые 2 файла, которые вы создали, в любую другую папку")
		echo "Для копирования  файлов используется следующая конструкция cp Имя файла.Расширение Имя файла2.Расширение  Путь куда переместить"
		echo "Давайте посмотрим есть ли что для копирования"
		ls -la
		echo "Введите 1ый файл в который хотите скопировать"
		read filename
		echo "Введите 2ый файл в который хотите скопировать"
		read filename1
		echo "Введите имя папки куда хотите скопировать файлы:"
		read dirname	
		cp $filename $filename1 $dirname
		echo "Проверим скопировали ли мы файлы? Перейдем в нужную папку"
		cd $dirname
		ls -la
		echo "Файлы тут? отлично"
		;;
	"Найти файл по имени")
		echo "Чтобы найти файл по имени используется следующая конструкция find Путь где искать -name название файла.расширение в двойных ковычках"
		echo "Введите 1ый файл в который хотите найти"
		read filename
		echo "Введите директорию где искать"
		read dirname
		find $dirname -name "$filename"
		;;
	"просмотреть содержимое в реальном времени")
		echo "Для поиска содержимого используется команда grep, у данной команды много ключей, мы же используем -in где i - отключает зависимость регистра n - нумерует строчку, где находит нужный файл"
		echo "Синтаксис такой grep -in Икомое выражение или условие Путь файла"
		echo "К примеру найдем в специально подготовленной книге по адресу D:/Gitbash/Homework1/book.txt заданый текст"
		echo "Какой текст вы хотите найти"
		read nt
		grep -in $nt D:/Gitbash/Homework1/book.txt
		;; 
	"вывести несколько первых строк из текстового файла")
		echo "Для того чтобы вывести первые строчки текстового файла используется команда head -nx, где х количество нужных строк. Путь файла"
		echo "Введите количество строк которые вы хотите вывести из файла"
		read number
		echo "Введите путь нужного текстового файла"
		read dir
		head -v -n$number $dir
		;; 
	"вывести несколько последних строк из текстового файла")
		echo "Для того чтобы вывести последние строчки текстового файла используется команда tail -nx, где х количество нужных строк. Путь файла"
		echo "Введите количество строк которые вы хотите вывести из файла"
		read number
		echo "Введите путь нужного текстового файла"
		read dir
		tail -v -n$number $dir
		;; 
	"просмотреть содержимое длинного файла")
		echo "если файл слишком большой и не помещается на страницу терминала, то можно использовать команду less ключи путь_файла"
		echo "К примеру откроем файл, для выхода из отдельного команда нажмите Q и Enter"
	
		echo "Введите путь нужного текстового файла"
		read dir
		
		less $dir
		;;
	"вывести дату и время")
		echo "Для вывода даты и времени используется команда date"
		date
		;;
	"Отправить http запрос на сервер")
		echo "Для отправки Http запроса используется команда curl адресс, мы так же используем ключ -k который игнорирует неподтвержденные сертификаты"
		echo "Запрос добавляем по адрессу из задания я и так не успеваю и устал"
		curl -k https://api.quarantine.country/api/v1/regions
		;;
        "Выход")
            break
            ;;
        *) echo "invalid option $REPLY";;
    esac
done
